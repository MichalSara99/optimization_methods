99 om_zero_order::powell_conjugate_method<fp_type>::minimize(
100  om_zero_order::f_vector_t<fp_type> objective,
101  om_zero_order::vector_arg_t<fp_type> const &init_guess) const {
102 
103  auto x_prev = init_guess;
104  std::size_t dim = init_guess.rows();
105  std::size_t iters{0};
106 
107  auto u = cartesian_basis_vectors<fp_type>()(dim);
108  auto x = std::vector<vector_t<fp_type>>(dim + 2);
109  x[0] = x_prev;
110  x[1] = x_prev;
111 
112  while (iters < this->max_iters_) {
113  for (auto t = 0; t < u.size(); ++t) {
114 
115  auto F_lambda = [&](fp_type const &lambda) {
116  return objective(x[t + 1] + lambda * u[t]);
117  };
118  auto optim_lambda = this->lsm_(F_lambda);
119  x[t + 2] = x[t + 1] + std::get<0>(optim_lambda) * u[t];
120  }
121 
122  for (auto t = 0; t < u.size() - 1; ++t) {
123  u[t] = u[t + 1];
124  }
125  u[u.size() - 1] = x.back() - x[1];
126 
127  if ((x.back() - x[1]).norm() <= this->converge_tol_) {
128  return std::make_tuple(x.back(), objective(x.back()), iters);
129  } else {
130  iters++;
131  x[0] = x.back();
132  }
133 
134  if ((iters % (dim + 1)) == 0) {
135  u = cartesian_basis_vectors<fp_type>()(dim);
136  x[1] = x.front();
137  }
138  }
139  return std::make_tuple(x.back(), objective(x.back()), iters);
140 }
