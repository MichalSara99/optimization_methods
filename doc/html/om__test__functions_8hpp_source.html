89  fp_type theta{0.0};
90  if (args(0) > static_cast<fp_type>(0.0)) {
91  theta =
92  (static_cast<fp_type>(1.0) / (static_cast<fp_type>(2.0) * pi<fp_type>)*(
93  std::atan(args(1) / args(0))));
94  } else {
95  theta = (static_cast<fp_type>(1.0) /
96  (static_cast<fp_type>(2.0) *
97  pi<fp_type>)*(pi<fp_type> + (std::atan(args(1) / args(0)))));
98  }
99  return (static_cast<fp_type>(100.0) *
100  (((args(2) - static_cast<fp_type>(10.0) * theta) *
101  (args(2) - static_cast<fp_type>(10.0) * theta)) +
102  ((std::sqrt(args(0) * args(0) + args(1) * args(1)) -
