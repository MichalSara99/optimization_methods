163 std::tuple<om_zero_order::vector_t<fp_type>, fp_type, std::size_t>
164 om_zero_order::nelder_mead_method<fp_type>::minimize(
165  f_vector_t<fp_type> objective,
166  vector_arg_t<fp_type> const &init_guess) const {
167 
168  std::size_t N = init_guess.rows();
169  std::size_t iters{0};
170  auto init_simplex =
171  random_vectors_from_guess<fp_type, std::uniform_real_distribution>()(
172  N + 1, init_guess);
173 
174  while (iters < this->max_iters_) {
175 
176  // 1.STEP: Sort:
177  std::sort(init_simplex.begin(), init_simplex.end(),
178  [&](vector_arg_t<fp_type> const &first,
179  vector_arg_t<fp_type> const &second) {
180  return (objective(first) < objective(second));
181  });
182  // Convergence check:
183  if ((init_simplex.front() - init_simplex.back()).norm() <=
184  this->converge_tol_) {
185  return std::make_tuple(init_simplex.front(),
186  objective(init_simplex.front()), iters);
187  }
188  // 2.STEP: Centroid:
189  vector_arg_t<fp_type> centroid = vector_arg_t<fp_type>::Zero(N);
190  for (auto t = 0; t < init_simplex.size() - 1; ++t) {
191  centroid += ((static_cast<fp_type>(1.0) / N) * init_simplex[t]);
192  }
193  // 3.STEP: Reflection:
194  vector_arg_t<fp_type> reflection = vector_arg_t<fp_type>::Zero(N);
195  reflection =
196  centroid + this->reflection_rho_ * (centroid - init_simplex.back());
197 
198  if ((objective(centroid) <= objective(reflection)) &&
199  (objective(reflection) <
200  objective(init_simplex[init_simplex.size() - 2]))) {
201  init_simplex[init_simplex.size() - 1] = reflection;
202  iters++;
203  // go to 1.STEP
204  continue;
205  } else if ((objective(reflection) < objective(centroid))) {
206  // 4.STEP: Expansion:
207  vector_arg_t<fp_type> expansion = vector_arg_t<fp_type>::Zero(N);
208  expansion =
209  centroid + this->expansion_rho_ * (centroid - init_simplex.back());
210 
211  if (objective(expansion) < objective(reflection)) {
212  init_simplex[init_simplex.size() - 1] = expansion;
213  } else {
214  init_simplex[init_simplex.size() - 1] = reflection;
215  }
216  iters++;
217  continue;
218  } else {
219  // 5.STEP: Contraction:
220  vector_arg_t<fp_type> contraction = vector_arg_t<fp_type>::Zero(N);
221  contraction =
222  centroid + this->contraction_rho_ * (centroid - init_simplex.back());
223 
224  if (objective(contraction) < objective(init_simplex.back())) {
225  init_simplex[init_simplex.size() - 1] = contraction;
226  iters++;
227  continue;
228  } else {
229  // 6.STEP: Shrinking:
230  for (auto t = 1; t < init_simplex.size(); ++t) {
231  init_simplex[t] =
232  (init_simplex.front() +
233  this->shrinkage_rho_ * (init_simplex[t] - init_simplex.front()));
234  }
235  iters++;
236  }
237  }
238  }
239  return std::make_tuple(init_simplex.front(), objective(init_simplex.front()),
240  iters);
241 }
