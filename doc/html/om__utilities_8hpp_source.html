210  fp_type y2) {
211  fp_type t1 = (x0 * y1 * y2 / ((y0 - y1) * (y0 - y2)));
212  fp_type t2 = (x1 * y0 * y2 / ((y1 - y0) * (y1 - y2)));
213  fp_type t3 = (x2 * y0 * y1 / ((y2 - y0) * (y2 - y1)));
214  return (t1 + t2 + t3);
