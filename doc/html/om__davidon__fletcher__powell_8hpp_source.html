80  // Start of Algorithm:
81  grad_prev = central_difference<1, fp_type>()(objective, x_prev);
82  u = static_cast<fp_type>(-1.0) * G * grad_prev;
83  while (iters < this->max_iters_) {
84 
85  auto F_lambda = [&](fp_type const &lambda) {
86  return objective(x_prev + lambda * u);
87  };
88  auto optim_lambda = this->lsm_(F_lambda);
89  x = x_prev + (std::get<0>(optim_lambda)) * u;
90  grad = central_difference<1, fp_type>()(objective, x);
91  // Stopping criteria:
92  if (((x - x_prev).norm() < this->arg_tol_) &&
93  (grad.norm() < this->grad_tol_) &&
94  (std::abs(objective(x) - objective(x_prev)) < this->fun_tol_)) {
95  return std::make_tuple(x, objective(x), iters);
96  } else {
97  v = (std::get<0>(optim_lambda)) * u;
98  y = grad - grad_prev;
99  // rank 1-update:
100  A = (v * v.transpose()) / (v.transpose() * y);
101  B = static_cast<fp_type>(-1.0) * (G * y * (G * y).transpose()) /
102  (y.transpose() * G * y);
103  // update:
104  G += (A + B);
105  grad_prev = grad;
106  x_prev = x;
107  u = static_cast<fp_type>(-1.0) * G * grad_prev;
108  iters++;
109  }
110  }
111  return std::make_tuple(x, objective(x), iters);
